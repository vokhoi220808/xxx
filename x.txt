UCLN

#include <bits/stdc++.h>
using namespace std;
const int N=1e5+7;
int a[N];
int UCLN(int a,int b){
    if(b==0){
        return a;
    }
    return UCLN(b,a%b);
}
int main(){
    int n;
    cin>>n;
    for(int i=1;i<=n;i++){
        cin>>a[i];
    }
    int res=0;
    for(int i=1;i<n;i++){
        res=UCLN(res,a[i]);
    }
    cout<<res;
    return 0;
}

FIBONACI



#include <bits/stdc++.h>
using namespace std;
long long f[1000000];
long long fib(int n){
    if(f[n]!=-1){
        return f[n];
    }
    if(n<=2){
        return f[n]=1;
    }
    return f[n]=fib(n-1)+fib(n-2);
}
int main(){
    int n;
    cin>>n;
    for(int i=0;i<1000000;i++){
        f[i]=-1;
    }
    cout<<fib(n);
}

#include<bits/stdc++.h>
using namespace std;

int x[20],used[20],n,k,dem=0;

void np(int i){
    if(i>k){
        dem++;
        cout<<dem<<". ";
        for(int j=1;j<=k;j++) cout<<x[j]<<" ";
        cout<<"\n";
        return;
    }
    for(int j=1;j<=n;j++){
        if(!used[j]){
            used[j]=1;
            x[i]=j;
            np(i+1);
            used[j]=0;
        }
    }
}

int main(){
    cin>>n>>k;
    np(1);
    return 0;
}

#include<bits/stdc++.h>
using namespace std;
int n,s,a[1001],t=0,dem=0;
void Try(int i){
    if(t==s){
        dem=1;
        return;
    }
    if(i>n||dem==1){
        return;
    }
    if(t+a[i]<=s){
        t+=a[i];
        Try(i+1);
        t-=a[i];
    }
    Try(i+1);
}
int main(){
    cin>>n>>s;
    for(int i=1;i<=n;i++){
        cin>>a[i];
    }
    Try(1);
    if(dem==0){
        cout<<-1;
    }else{
        cout<<1;
    }
}
