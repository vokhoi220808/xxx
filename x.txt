UCLN

#include <bits/stdc++.h>
using namespace std;
const int N=1e5+7;
int a[N];
int UCLN(int a,int b){
    if(b==0){
        return a;
    }
    return UCLN(b,a%b);
}
int main(){
    int n;
    cin>>n;
    for(int i=1;i<=n;i++){
        cin>>a[i];
    }
    int res=0;
    for(int i=1;i<n;i++){
        res=UCLN(res,a[i]);
    }
    cout<<res;
    return 0;
}

#include <bits/stdc++.h>
using namespace std;
long long f[1000000];
long long fib(int n){
    if(f[n]!=-1){
        return f[n];
    }
    if(n<=2){
        return f[n]=1;
    }
    return f[n]=fib(n-1)+fib(n-2);
}
int main(){
    int n;
    cin>>n;
    for(int i=0;i<1000000;i++){
        f[i]=-1;
    }
    cout<<fib(n);
}

#include<bits/stdc++.h>
using namespace std;
int n,s,a[1001],t=0,dem=0;
void Try(int i){
    if(t==s){
        dem=1;
        return;
    }
    if(i>n||dem==1){
        return;
    }
    if(t+a[i]<=s){
        t+=a[i];
        Try(i+1);
        t-=a[i];
    }
    Try(i+1);
}
int main(){
    cin>>n>>s;
    for(int i=1;i<=n;i++){
        cin>>a[i];
    }
    Try(1);
    if(dem==0){
        cout<<-1;
    }else{
        cout<<1;
    }
}

#include <iostream>

int n;
char X[20];
void Try(int i) {
    X[i] = 'A';
    if (i == n - 1) {
        for (int j = 0; j < n; j++) cout << X[j];
        cout << endl;
    } else {
        Try(i + 1);
    }
    if (i == 0 || X[i - 1] != 'B') {
        X[i] = 'B';
        if (i == n - 1) {
            for (int j = 0; j < n; j++) cout << X[j];
            cout << endl;
        } else {
            Try(i + 1);
        }
    }
}

int main() {
    cin >> n;
    Try(0);
    return 0;
}

#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, W;
    cin >> n >> W;

    vector<int> w(n+1), v(n+1);
    for (int i = 1; i <= n; i++) {
        cin >> w[i] >> v[i];
    }
    vector<int> dp(W+1, 0);

    for (int i = 1; i <= n; i++) {
        for (int j = W; j >= w[i]; j--) {
            dp[j] = max(dp[j], dp[j - w[i]] + v[i]);
        }
    }

    cout << dp[W] << "\n";

    return 0;
}

#include <bits/stdc++.h>
using namespace std;
int n;
int countWays = 0;
bool col[12];     
bool diag1[23];   
bool diag2[23];      

void backtrack(int row) {
    if (row == n) {
        countWays++;
        return;
    }
    for (int c = 0; c < n; c++) {
        if (!col[c] && !diag1[row+c] && !diag2[row-c+n-1]) {
            col[c] = diag1[row+c] = diag2[row-c+n-1] = true;
            backtrack(row + 1);
            col[c] = diag1[row+c] = diag2[row-c+n-1] = false;
        }
    }
}
int main() {
    cin >> n;
    for(int i=0;i<12;i++) col[i] = false;
    for(int i=0;i<23;i++) diag1[i] = diag2[i] = false;
    backtrack(0);
    cout << countWays << endl;
    return 0;
}
